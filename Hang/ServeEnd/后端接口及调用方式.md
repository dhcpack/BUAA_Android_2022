// com.example.hang.ports.HttpUtil.java
// 实现了基本的GET, POST, PUT, DELETE方法
// 每个方法内新开子线程访问后端接口（安卓要求网络请求需要开辟子线程）
/*
 * GET: isArray=True返回JsonArray，isArray=False返回JsonObject(这块好像有点bug)
 * */
public static Object httpGet(String url, ArrayList<String> params, boolean isArray){}

/*
 * POST
 * */
public static JSONObject httpPost(String url, HashMap<String, String> params)
    
/*
 * PUT
 * */
public static JSONObject httpPut(String url, HashMap<String, String> params)
    
/*
 * DELETE
 * */
public static JSONObject httpDelete(String url, ArrayList<String> params)
